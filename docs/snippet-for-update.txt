
Update could be _inside_ commander.
But better outside since the last step is for cyber-dojo to replace itself.
[./cyber-dojo update] is the only way to change what :latest is


#!/bin/bash
set -e
shift # update

# - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
#  Use: cyber-dojo update [latest|RELEASE]
#
#  Example 1: change to latest (the default)
#
#    cyber-dojo update
#    cyber-dojo update latest
#
#    Pulls latest ready for the next [cyber-dojo up] command.
#    Tags it with its release number (eg 3.19.1)
#
#  Example 2: change to a given release
#
#    cyber-dojo update 2.7.31
#
#    Pulls 2.7.31 and tags it with latest
#    ready for the next [cyber-dojo up] command.
#
#  Example 3: change to a given tag
#
#    cyber-dojo update 3ec864c
#
#    Pulls 3ec864c and tags it with latest
#    ready for the next [cyber-dojo up] command.
#    Warns that you are on a development tag.
# - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

# keep a release-tag for the outgoing latest
readonly WAS_VERSIONER_IMAGE=cyberdojo/versioner:latest
readonly WAS=$(docker run --rm ${WAS_VERSIONER_IMAGE} sh -c 'echo -n ${RELEASE}')
if [ -n "${WAS}" ]; then
  docker tag ${WAS_VERSIONER_IMAGE} cyberdojo/versioner:${WAS}
else
  readonly WAS_SHA=$(docker run --rm ${WAS_VERSIONER_IMAGE} sh -c 'echo -n ${SHA}')
  readonly WAS_TAG=${WAS_SHA:0:7}
  docker tag ${WAS_VERSIONER_IMAGE} cyberdojo/versioner:${WAS_TAG}
fi

readonly RELEASE=${1:-latest}
readonly NOW_VERSIONER_IMAGE=cyberdojo/versioner:${RELEASE}
docker pull ${NOW_VERSIONER_IMAGE}
docker tag ${NOW_VERSIONER_IMAGE} cyberdojo/versioner:latest

# Do this? Not symmetric with keeping release-tag above.
# Can still get release tag and display it when any command is run.
# Maybe this should be done via the command:
# ./cyber-dojo version
readonly NOW_TAG=$(docker run --rm ${NOW_VERSIONER_IMAGE} sh -c 'echo -n ${RELEASE}')
if [ "${RELEASE}" = 'latest' ]; then
  docker tag ${NOW_VERSIONER_IMAGE} cyberdojo/versioner:${NOW_TAG}
elif [ -z "${NOW_TAG}" ]; then
  echo "Warning: ${RELEASE} is dev only"
end



# - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
All commands always use versioner:latest to get commander's tag
Update is the only way you change versioner:latest

# The two tag commands are because you can do
# 1) ./cyber-dojo update 2.3.0
# 2) ./cyber-dojo update latest
# And if we do (2) we need to create a non-latest tag
# because latest moves and we want to keep a local history.
